            function verifyCookie(c_name, cpar) {
                var name = c_name + "=";
                var ca = document.cookie.split(';');
                for (var i = 0; i < ca.length; i++) {
                    var c = ca[i];
                    while (c.charAt(0) == ' ') {
                        c = c.substring(1);
                    }
                    if (c.indexOf(name) == 0 && cpar == undefined) {
                        return c.substring(name.length, c.length);

                    if (c.indexOf(name) == 0 && cpar) {
